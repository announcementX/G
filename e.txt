-- 绑定按钮事件
authorBtn.MouseButton1Click:Connect(showAuthor)
themeBtn.MouseButton1Click:Connect(showTheme)
minimizeBtn.MouseButton1Click:Connect(function()
	log("缩小按钮被点击")
	-- 缩小为圆角正方形
	local dot = Instance.new("Frame")
	dot.Size = UDim2.new(0,50,0,50)
	local center = mainFrame.AbsolutePosition + mainFrame.AbsoluteSize/2
	dot.Position = UDim2.new(0, center.X-25, 0, center.Y-25)
	dot.BackgroundColor3 = Color3.fromRGB(0,160,255)
	dot.Parent = screenGui
	local dotCorner = Instance.new("UICorner")
	dotCorner.CornerRadius = UDim.new(0,12)
	dotCorner.Parent = dot
	local xu = Instance.new("TextLabel")
	xu.Size = UDim2.new(1,0,1,0)
	xu.BackgroundTransparency = 1
	xu.Text = "XU"
	xu.TextColor3 = Color3.fromRGB(255,255,255)
	xu.TextSize = 20
	xu.Font = Enum.Font.GothamBold
	xu.Parent = dot
	mainFrame.Visible = false
	
	-- 长按2秒拖动
	local dragging = false
	local dragStart
	local longPressTimer
	local function onInputBegan(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 then
			dragStart = input.Position
			longPressTimer = task.delay(2, function()
				dragging = true
			end)
		end
	end
	local function onInputChanged(input)
		if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
			local delta = input.Position - dragStart
			dot.Position = UDim2.new(0, dot.Position.X.Offset + delta.X, 0, dot.Position.Y.Offset + delta.Y)
			dragStart = input.Position
		end
	end
	local function onInputEnded(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 then
			if longPressTimer then task.cancel(longPressTimer) end
			if not dragging then
				-- 单击恢复窗口
				mainFrame.Visible = true
				dot:Destroy()
				log("恢复窗口")
			end
			dragging = false
		end
	end
	dot.InputBegan:Connect(onInputBegan)
	UserInputService.InputChanged:Connect(onInputChanged)
	UserInputService.InputEnded:Connect(onInputEnded)
end)

-- 窗口拖动
local dragData = { active = false, startPos = nil, startMouse = nil }
dragHandle.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 then
		dragData.active = true
		dragData.startPos = mainFrame.Position
		dragData.startMouse = input.Position
	end
end)
UserInputService.InputChanged:Connect(function(input)
	if dragData.active and input.UserInputType == Enum.UserInputType.MouseMovement then
		local delta = input.Position - dragData.startMouse
		mainFrame.Position = UDim2.new(0, dragData.startPos.X.Offset + delta.X, 0, dragData.startPos.Y.Offset + delta.Y)
	end
end)
UserInputService.InputEnded:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 then
		dragData.active = false
	end
end)
