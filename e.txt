-- 长按拖动逻辑 (通用)
local dragging = false
local dragStartPos = nil
local dragFrame = nil
local longPressTimer = nil
local isLongPressing = false
local longPressThreshold = 0.5  -- 0.5秒长按

local function startDrag(frame, input)
    if dragging then return end
    dragging = true
    dragFrame = frame
    dragStartPos = Vector2.new(input.Position.X - frame.AbsolutePosition.X, input.Position.Y - frame.AbsolutePosition.Y)
end

local function updateDrag(input)
    if not dragging or not dragFrame then return end
    local newX = input.Position.X - dragStartPos.X
    local newY = input.Position.Y - dragStartPos.Y
    -- 边界限制 (防止拖出屏幕)
    local maxX = game.CoreGui.AbsoluteSize.X - dragFrame.AbsoluteSize.X
    local maxY = game.CoreGui.AbsoluteSize.Y - dragFrame.AbsoluteSize.Y
    newX = math.clamp(newX, 0, maxX)
    newY = math.clamp(newY, 0, maxY)
    dragFrame.Position = UDim2.new(0, newX, 0, newY)
end

local function stopDrag()
    dragging = false
    dragFrame = nil
    dragStartPos = nil
end

-- 长按检测
local function onInputBegan(input, gameProcessed)
    if gameProcessed then return end
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        -- 检查点击是否在可拖动区域内 (标题栏 或 小方块)
        local hitFrame = nil
        if mainFrame.Visible and input.UserInputType == Enum.UserInputType.MouseButton1 then
            local mousePos = input.Position
            if mousePos.X >= titleBar.AbsolutePosition.X and mousePos.X <= titleBar.AbsolutePosition.X + titleBar.AbsoluteSize.X and
               mousePos.Y >= titleBar.AbsolutePosition.Y and mousePos.Y <= titleBar.AbsolutePosition.Y + titleBar.AbsoluteSize.Y then
                hitFrame = mainFrame
            end
        elseif miniFrame.Visible then
            local mousePos = input.Position
            if mousePos.X >= miniFrame.AbsolutePosition.X and mousePos.X <= miniFrame.AbsolutePosition.X + miniFrame.AbsoluteSize.X and
               mousePos.Y >= miniFrame.AbsolutePosition.Y and mousePos.Y <= miniFrame.AbsolutePosition.Y + miniFrame.AbsoluteSize.Y then
                hitFrame = miniFrame
            end
        end
        
        if hitFrame then
            longPressTimer = tick()
            isLongPressing = true
            -- 等待长按触发
            spawn(function()
                while isLongPressing and (tick() - longPressTimer) < longPressThreshold do
                    wait()
                end
                if isLongPressing and (tick() - longPressTimer) >= longPressThreshold then
                    startDrag(hitFrame, input)
                end
            end)
        end
    end
end

local function onInputEnded(input, gameProcessed)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        isLongPressing = false
        stopDrag()
    end
end

local function onInputChanged(input, gameProcessed)
    if gameProcessed then return end
    if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        updateDrag(input)
    end
end

UserInputService.InputBegan:Connect(onInputBegan)
UserInputService.InputEnded:Connect(onInputEnded)
UserInputService.InputChanged:Connect(onInputChanged)

-- 更新时间 (实时刷新)
local function updateClock()
    while true do
        local t = os.date("*t")
        local timeStr = string.format("北京时间 %02d:%02d:%02d", t.hour, t.min, t.sec)
        clockLabel.Text = timeStr
        wait(0.2)
    end
end
