    悬浮窗脚本 - 可移动悬浮窗，支持缩小为圆点，长按/拖动圆点移动

-- ==================== 拖拽逻辑（悬浮窗通过标题栏拖动） ====================
local windowDragData = {
    dragging = false,
    startPos = nil,
    startMousePos = nil,
}

titleBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        windowDragData.dragging = true
        windowDragData.startPos = floatingWindow.Position
        windowDragData.startMousePos = input.Position
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                windowDragData.dragging = false
            end
        end)
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if windowDragData.dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        local delta = input.Position - windowDragData.startMousePos
        local newXOffset = windowDragData.startPos.X.Offset + delta.X
        local newYOffset = windowDragData.startPos.Y.Offset + delta.Y
        floatingWindow.Position = UDim2.new(0, newXOffset, 0, newYOffset)
    end
end)

UserInputService.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        windowDragData.dragging = false
    end
end)

-- ==================== 圆点拖动逻辑（带点击/拖动区分） ====================
local dotDragData = {
    active = false,      -- 是否处于拖动模式
    dragging = false,    -- 是否正在拖动中（移动距离超过阈值）
    startPos = nil,
    startMousePos = nil,
    connectionInputChanged = nil,
    connectionInputEnded = nil,
}

-- 清理圆点的事件连接
local function cleanupDotDrag()
    if dotDragData.connectionInputChanged then
        dotDragData.connectionInputChanged:Disconnect()
        dotDragData.connectionInputChanged = nil
    end
    if dotDragData.connectionInputEnded then
        dotDragData.connectionInputEnded:Disconnect()
        dotDragData.connectionInputEnded = nil
    end
    dotDragData.active = false
    dotDragData.dragging = false
end

-- 开始监听圆点拖动
local function startDotDrag(startInput)
    if dotDragData.active then return end
    dotDragData.active = true
    dotDragData.dragging = false
    dotDragData.startPos = dot.Position
    dotDragData.startMousePos = startInput.Position
    
    -- 监听移动
    dotDragData.connectionInputChanged = UserInputService.InputChanged:Connect(function(input)
        if not dotDragData.active then return end
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            local delta = input.Position - dotDragData.startMousePos
            local distance = math.sqrt(delta.X^2 + delta.Y^2)
            
            -- 超过阈值，进入拖动模式
            if not dotDragData.dragging and distance >= dragThreshold then
                dotDragData.dragging = true
            end
            
            -- 拖动模式下更新圆点位置
            if dotDragData.dragging then
                local newXOffset = dotDragData.startPos.X.Offset + delta.X
                local newYOffset = dotDragData.startPos.Y.Offset + delta.Y
                dot.Position = UDim2.new(0, newXOffset, 0, newYOffset)
            end
        end
    end)
    
    -- 监听结束
    dotDragData.connectionInputEnded = UserInputService.InputEnded:Connect(function(input)
        if not dotDragData.active then return end
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            -- 如果没有进入拖动模式，则视为点击（恢复悬浮窗）
            if not dotDragData.dragging then
                -- 恢复悬浮窗
                restoreWindowFromDot()
            end
            cleanupDotDrag()
        end
    end)
end

-- 圆点按下事件
dot.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        startDotDrag(input)
    end
end)

-- ==================== 辅助函数 ====================
-- 获取控件的中心点绝对坐标（像素）
local function getCenterPosition(guiObject)
    local absPos = guiObject.AbsolutePosition
    local absSize = guiObject.AbsoluteSize
    return Vector2.new(absPos.X + absSize.X/2, absPos.Y + absSize.Y/2)
end

-- 根据中心点坐标设置控件位置（基于左上角偏移）
local function setPositionFromCenter(guiObject, centerPos)
    local size = guiObject.AbsoluteSize
    local newPos = Vector2.new(centerPos.X - size.X/2, centerPos.Y - size.Y/2)
    guiObject.Position = UDim2.new(0, newPos.X, 0, newPos.Y)
end

-- 将悬浮窗缩小化为圆点
local function minimizeToDot()
    -- 获取悬浮窗中心点
    local windowCenter = getCenterPosition(floatingWindow)
    -- 隐藏悬浮窗，显示圆点
    floatingWindow.Visible = false
    dot.Visible = true
    -- 将圆点中心对齐到悬浮窗中心
    setPositionFromCenter(dot, windowCenter)
    -- 添加淡入动画效果
    dot.BackgroundTransparency = 0.15
end

-- 从圆点恢复悬浮窗
local function restoreWindowFromDot()
    -- 获取圆点中心点
    local dotCenter = getCenterPosition(dot)
    -- 显示悬浮窗，隐藏圆点
    floatingWindow.Visible = true
    dot.Visible = false
    -- 将悬浮窗中心对齐到圆点中心
    setPositionFromCenter(floatingWindow, dotCenter)
    -- 添加轻微的弹入效果（可选）
    floatingWindow.BackgroundTransparency = 0.1
end
