-- 应用样式函数
local function applyStyle(styleIndex)
    local style = styles[styleIndex]
    if not style then return end
    
    -- 应用颜色
    mainFrame.BackgroundColor3 = style.mainBg
    sidebar.BackgroundColor3 = style.sidebarBg
    titleBar.BackgroundColor3 = style.titleBg
    contentArea.BackgroundColor3 = style.contentBg
    
    -- 更新所有侧边栏按钮颜色
    for _, btn in pairs(tabButtons) do
        if btn then
            btn.BackgroundColor3 = style.sidebarBg
            btn.TextColor3 = style.textColor
        end
    end
    
    -- 更新标题栏文字颜色
    titleLabel.TextColor3 = style.textColor
    clockLabel.TextColor3 = style.textColor
    
    -- 更新信息面板文字颜色
    infoTitle.TextColor3 = style.textColor
    authorLine.TextColor3 = Color3.fromRGB(
        style.textColor.R * 0.8,
        style.textColor.G * 0.8,
        style.textColor.B * 0.8
    )
    qqLine.TextColor3 = authorLine.TextColor3
    descLine.TextColor3 = Color3.fromRGB(
        style.textColor.R * 0.6,
        style.textColor.G * 0.6,
        style.textColor.B * 0.6
    )
    
    -- 更新样式面板
    styleTitle.TextColor3 = style.textColor
    for i, btn in ipairs(styleBtns) do
        if btn then
            if i == styleIndex then
                btn.BackgroundColor3 = style.accentColor
                btn.TextColor3 = Color3.fromRGB(255, 255, 255)
            else
                btn.BackgroundColor3 = Color3.fromRGB(
                    style.sidebarBg.R * 1.2,
                    style.sidebarBg.G * 1.2,
                    style.sidebarBg.B * 1.2
                )
                btn.TextColor3 = style.textColor
            end
        end
    end
    
    currentStyle = styleIndex
end

-- ========== 拖拽系统（参考Roblox开发者论坛拖拽GUI教程[citation:5]）==========
local dragging = false
local dragFrame = nil
local dragStart = nil
local startPos = nil
local longPressTimer = nil
local isLongPressing = false
local LONG_PRESS_TIME = 0.3

-- 更新拖拽位置
local function updateDrag(input)
    if not dragging or not dragFrame then return end
    local delta = input.Position - dragStart
    local newX = startPos.X.Offset + delta.X
    local newY = startPos.Y.Offset + delta.Y
    -- 边界限制
    local maxX = game.CoreGui.AbsoluteSize.X - dragFrame.AbsoluteSize.X
    local maxY = game.CoreGui.AbsoluteSize.Y - dragFrame.AbsoluteSize.Y
    newX = math.clamp(newX, 0, maxX)
    newY = math.clamp(newY, 0, maxY)
    dragFrame.Position = UDim2.new(0, newX, 0, newY)
end

-- 开始拖拽
local function beginDrag(frame, input)
    if dragging then return end
    dragging = true
    dragFrame = frame
    dragStart = input.Position
    startPos = frame.Position
end

-- 结束拖拽
local function endDrag()
    dragging = false
    dragFrame = nil
    dragStart = nil
    startPos = nil
end

-- 标题栏拖拽（普通点击拖拽）
titleBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        beginDrag(mainFrame, input)
    end
end)

-- 小方块长按拖拽
local miniLongPressActive = false
miniFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        longPressTimer = tick()
        isLongPressing = true
        task.spawn(function()
            while isLongPressing and (tick() - longPressTimer) < LONG_PRESS_TIME do
                task.wait()
            end
            if isLongPressing and (tick() - longPressTimer) >= LONG_PRESS_TIME then
                miniLongPressActive = true
                beginDrag(miniFrame, input)
            end
        end)
    end
end)

miniFrame.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        isLongPressing = false
        -- 如果不是长按拖拽，则视为单击恢复窗口
        if not miniLongPressActive and miniFrame.Visible then
            mainFrame.Visible = true
            miniFrame.Visible = false
            mainFrame.Position = UDim2.new(0, miniFrame.AbsolutePosition.X, 0, miniFrame.AbsolutePosition.Y)
        end
        miniLongPressActive = false
        endDrag()
    end
end)

-- 全局拖拽更新
UserInputService.InputChanged:Connect(function(input)
    if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        updateDrag(input)
    end
end)

-- 确保松手时结束拖拽
UserInputService.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        endDrag()
    end
end)

-- ========== 北京时间更新 ==========
local function updateClock()
    while mainGui and mainGui.Parent do
        local t = os.date("*t")
        local timeStr = string.format("%02d:%02d:%02d", t.hour, t.min, t.sec)
        clockLabel.Text = timeStr
        task.wait(0.2)
    end
end
task.spawn(updateClock)
