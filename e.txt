-- ========== 拖拽系统（主窗口）==========
local dragActive = false
local dragStart = nil
local startPos = nil
titleBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragActive = true
        dragStart = input.Position
        startPos = mainFrame.Position
    end
end)
UserInputService.InputChanged:Connect(function(input)
    if dragActive and input.UserInputType == Enum.UserInputType.MouseMovement then
        local delta = input.Position - dragStart
        local newX = startPos.X.Offset + delta.X
        local newY = startPos.Y.Offset + delta.Y
        local maxX = game.CoreGui.AbsoluteSize.X - mainFrame.AbsoluteSize.X
        local maxY = game.CoreGui.AbsoluteSize.Y - mainFrame.AbsoluteSize.Y
        mainFrame.Position = UDim2.new(0, math.clamp(newX,0,maxX), 0, math.clamp(newY,0,maxY))
    end
end)
UserInputService.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragActive = false
    end
end)

-- 小方块拖拽
local miniDragActive = false
local miniDragStart = nil
local miniStartPos = nil
local longPressTimer = nil

miniFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        longPressTimer = task.delay(0.35, function()
            miniDragActive = true
            miniDragStart = input.Position
            miniStartPos = miniFrame.Position
        end)
    end
end)

miniFrame.InputEnded:Connect(function(input)
    if longPressTimer then
        longPressTimer:Cancel()
        longPressTimer = nil
    end
    if miniDragActive then
        miniDragActive = false
    elseif miniFrame.Visible then
        -- 单击恢复主窗口
        mainFrame.Visible = true
        miniFrame.Visible = false
        mainFrame.Position = UDim2.new(0, miniFrame.AbsolutePosition.X, 0, miniFrame.AbsolutePosition.Y)
    end
end)

-- 全局更新小方块拖拽
UserInputService.InputChanged:Connect(function(input)
    if miniDragActive and input.UserInputType == Enum.UserInputType.MouseMovement then
        local delta = input.Position - miniDragStart
        local newX = miniStartPos.X.Offset + delta.X
        local newY = miniStartPos.Y.Offset + delta.Y
        local maxX = game.CoreGui.AbsoluteSize.X - miniFrame.AbsoluteSize.X
        local maxY = game.CoreGui.AbsoluteSize.Y - miniFrame.AbsoluteSize.Y
        miniFrame.Position = UDim2.new(0, math.clamp(newX,0,maxX), 0, math.clamp(newY,0,maxY))
    end
end)

-- ========== 应用完整样式（颜色+布局）==========
local function applyFullStyle()
    local style = stylesData[currentStyle]
    if not style then return end
    
    -- 应用颜色
    mainFrame.BackgroundColor3 = style.mainBg
    sidebar.BackgroundColor3 = style.sidebarBg
    titleBar.BackgroundColor3 = style.titleBg
    contentArea.BackgroundColor3 = style.contentBg
    minBtn.BackgroundColor3 = style.sidebarBg
    titleLabel.TextColor3 = style.textColor
    clockLabel.TextColor3 = style.textColor
    
    -- 重建面板（颜色跟随）
    buildInfoPanel(style)
    buildStylePanel()
    
    -- 应用布局
    local sidebarWidth = 100
    if style.layout == "left" then
        sidebar.Size = UDim2.new(0, sidebarWidth, 1, -32)
        sidebar.Position = UDim2.new(0, 0, 0, 32)
        sidebarList.Size = UDim2.new(1,0,1,0)
        listLayout.FillDirection = Enum.FillDirection.Vertical
        listLayout.Padding = UDim.new(0,6)
        contentArea.Size = UDim2.new(1, -sidebarWidth, 1, -32)
        contentArea.Position = UDim2.new(0, sidebarWidth, 0, 32)
        infoPanel.Size = UDim2.new(1,0,1,0)
        stylePanel.Size = UDim2.new(1,0,1,0)
    elseif style.layout == "right" then
        sidebar.Size = UDim2.new(0, sidebarWidth, 1, -32)
        sidebar.Position = UDim2.new(1, -sidebarWidth, 0, 32)
        sidebarList.Size = UDim2.new(1,0,1,0)
        listLayout.FillDirection = Enum.FillDirection.Vertical
        listLayout.Padding = UDim.new(0,6)
        contentArea.Size = UDim2.new(1, -sidebarWidth, 1, -32)
        contentArea.Position = UDim2.new(0, 0, 0, 32)
        infoPanel.Size = UDim2.new(1,0,1,0)
        stylePanel.Size = UDim2.new(1,0,1,0)
    elseif style.layout == "top" then
        sidebar.Size = UDim2.new(1,0,0,50)
        sidebar.Position = UDim2.new(0,0,0,32)
        sidebarList.Size = UDim2.new(1,0,1,0)
        listLayout.FillDirection = Enum.FillDirection.Horizontal
        listLayout.HorizontalAlignment = Enum.HorizontalAlignment.Center
        listLayout.Padding = UDim.new(0,15)
        contentArea.Size = UDim2.new(1,0,1,-82)
        contentArea.Position = UDim2.new(0,0,0,82)
        infoPanel.Size = UDim2.new(1,0,1,0)
        stylePanel.Size = UDim2.new(1,0,1,0)
    end
    
    -- 更新侧边栏按钮样式
    for id,btn in pairs(tabButtons) do
        if btn then
            btn.BackgroundColor3 = style.sidebarBg
            btn.TextColor3 = style.textColor
            if selectedTab == id then
                btn.BackgroundColor3 = style.accent
            end
        end
    end
    
    -- 高亮当前样式按钮
    for i,btn in ipairs(styleBtns) do
        if btn then
            TweenService:Create(btn, TweenInfo.new(0.2), {
                BackgroundColor3 = (i == currentStyle) and style.accent or style.cardBg
            }):Play()
        end
    end
end

-- ========== 构建侧边栏栏目按钮 ==========
local function buildSidebarButtons()
    for _,btn in pairs(tabButtons) do if btn then btn:Destroy() end end
    tabButtons = {}
    local style = stylesData[currentStyle]
    for _,tab in ipairs(tabs) do
        local btn = Instance.new("TextButton")
        btn.Size = UDim2.new(0.85,0,0,38)
        btn.BackgroundColor3 = style.sidebarBg
        btn.Text = tab.name
        btn.TextColor3 = style.textColor
        btn.TextSize = 14
        btn.Font = Enum.Font.Gotham
        btn.BorderSizePixel = 0
        btn.Parent = sidebarList
        local corner = Instance.new("UICorner")
        corner.CornerRadius = UDim.new(0,6)
        corner.Parent = btn
        tabButtons[tab.id] = btn
        btn.MouseButton1Click:Connect(function()
            selectedTab = tab.id
            infoPanel.Visible = (tab.id == "info")
            stylePanel.Visible = (tab.id == "style")
            for id,b in pairs(tabButtons) do
                if b then
                    b.BackgroundColor3 = (id == selectedTab) and style.accent or style.sidebarBg
                end
            end
        end)
        btn.MouseEnter:Connect(function()
            if selectedTab ~= tab.id then
                TweenService:Create(btn, TweenInfo.new(0.15), {BackgroundColor3 = style.hover}):Play()
            end
        end)
        btn.MouseLeave:Connect(function()
            if selectedTab ~= tab.id then
                TweenService:Create(btn, TweenInfo.new(0.15), {BackgroundColor3 = style.sidebarBg}):Play()
            end
        end)
    end
    if tabButtons["info"] then
        tabButtons["info"].BackgroundColor3 = style.accent
    end
end
