-- ========== 拖拽系统 ==========
local dragActive = false
local dragStartPos = nil
local dragFrameStartPos = nil
local longPressTimer = nil
local isLongPressing = false

local function startMainDrag(input)
    dragActive = true
    dragStartPos = input.Position
    dragFrameStartPos = mainFrame.Position
end

local function updateMainDrag(input)
    if not dragActive then return end
    local delta = input.Position - dragStartPos
    local newX = dragFrameStartPos.X.Offset + delta.X
    local newY = dragFrameStartPos.Y.Offset + delta.Y
    local maxX = game.CoreGui.AbsoluteSize.X - mainFrame.AbsoluteSize.X
    local maxY = game.CoreGui.AbsoluteSize.Y - mainFrame.AbsoluteSize.Y
    mainFrame.Position = UDim2.new(0, math.clamp(newX, 0, maxX), 0, math.clamp(newY, 0, maxY))
end

titleBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        startMainDrag(input)
    end
end)

local function startMiniDrag(input)
    miniDragActive = true
    miniDragStartPos = input.Position
    miniDragFrameStartPos = miniFrame.Position
end

local function updateMiniDrag(input)
    if not miniDragActive then return end
    local delta = input.Position - miniDragStartPos
    local newX = miniDragFrameStartPos.X.Offset + delta.X
    local newY = miniDragFrameStartPos.Y.Offset + delta.Y
    local maxX = game.CoreGui.AbsoluteSize.X - miniFrame.AbsoluteSize.X
    local maxY = game.CoreGui.AbsoluteSize.Y - miniFrame.AbsoluteSize.Y
    miniFrame.Position = UDim2.new(0, math.clamp(newX, 0, maxX), 0, math.clamp(newY, 0, maxY))
end

miniFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        isLongPressing = true
        longPressTimer = task.delay(0.35, function()
            if isLongPressing then
                startMiniDrag(input)
            end
        end)
    end
end)

miniFrame.InputEnded:Connect(function(input)
    if longPressTimer then
        longPressTimer:Cancel()
        longPressTimer = nil
    end
    if miniDragActive then
        miniDragActive = false
    elseif isLongPressing and miniFrame.Visible then
        mainFrame.Visible = true
        miniFrame.Visible = false
        mainFrame.Position = UDim2.new(0, miniFrame.AbsolutePosition.X, 0, miniFrame.AbsolutePosition.Y)
    end
    isLongPressing = false
end)

UserInputService.InputChanged:Connect(function(input)
    if dragActive and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        updateMainDrag(input)
    elseif miniDragActive and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        updateMiniDrag(input)
    end
end)

UserInputService.InputEnded:Connect(function()
    dragActive = false
    miniDragActive = false
end)

-- ========== 应用样式 ==========
local function applyStyle()
    local style = stylesData[currentStyleIndex]
    if not style then return end
    
    mainFrame.BackgroundColor3 = style.mainBg
    sidebar.BackgroundColor3 = style.sidebarBg
    titleBar.BackgroundColor3 = style.titleBg
    contentArea.BackgroundColor3 = style.contentBg
    minBtn.BackgroundColor3 = style.sidebarBg
    titleLabel.TextColor3 = style.textColor
    clockLabel.TextColor3 = style.textColor
    
    buildInfoPanel(style)
    buildStylePanel()
    
    for id, btn in pairs(tabButtons) do
        if btn then
            btn.BackgroundColor3 = style.sidebarBg
            btn.TextColor3 = style.textColor
            if selectedTab == id then
                btn.BackgroundColor3 = style.accentColor
            end
        end
    end
    
    for i, btn in ipairs(styleButtons) do
        if btn then
            TweenService:Create(btn, TweenInfo.new(0.2), {
                BackgroundColor3 = (i == currentStyleIndex) and style.accentColor or style.cardBg
            }):Play()
        end
    end
end

-- ========== 构建侧边栏按钮 ==========
local function buildSidebarButtons()
    for _, btn in pairs(tabButtons) do
        if btn then btn:Destroy() end
    end
    tabButtons = {}
    
    local style = stylesData[currentStyleIndex]
    
    for _, tab in ipairs(tabs) do
        local btn = Instance.new("TextButton")
        btn.Size = UDim2.new(0.85, 0, 0, 45)
        btn.BackgroundColor3 = style.sidebarBg
        btn.Text = tab.name
        btn.TextColor3 = style.textColor
        btn.TextSize = 15
        btn.Font = Enum.Font.Gotham
        btn.BorderSizePixel = 0
        btn.Parent = sidebarList
        local corner = Instance.new("UICorner")
        corner.CornerRadius = UDim.new(0, 8)
        corner.Parent = btn
        
        tabButtons[tab.id] = btn
        
        btn.MouseButton1Click:Connect(function()
            selectedTab = tab.id
            infoPanel.Visible = (tab.id == "info")
            stylePanel.Visible = (tab.id == "style")
            for id, b in pairs(tabButtons) do
                if b then
                    b.BackgroundColor3 = (id == selectedTab) and style.accentColor or style.sidebarBg
                end
            end
        end)
        
        btn.MouseEnter:Connect(function()
            if selectedTab ~= tab.id then
                TweenService:Create(btn, TweenInfo.new(0.15), {BackgroundColor3 = style.hoverColor}):Play()
            end
        end)
        btn.MouseLeave:Connect(function()
            if selectedTab ~= tab.id then
                TweenService:Create(btn, TweenInfo.new(0.15), {BackgroundColor3 = style.sidebarBg}):Play()
            end
        end)
    end
    
    if tabButtons["info"] then
        tabButtons["info"].BackgroundColor3 = style.accentColor
    end
end
