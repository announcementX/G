-- ========== 创建样式面板（卡片式按钮） ==========
local function rebuildStylePanel()
    for _, child in ipairs(stylePanel:GetChildren()) do
        child:Destroy()
    end
    
    local currentStyle = stylesData[currentStyleIndex]
    local cardColor = currentStyle.cardBg
    local textColor = currentStyle.textColor
    
    local titleCard = Instance.new("Frame")
    titleCard.Size = UDim2.new(1, -20, 0, 45)
    titleCard.Position = UDim2.new(0, 10, 0, 10)
    titleCard.BackgroundColor3 = cardColor
    titleCard.BorderSizePixel = 0
    titleCard.Parent = stylePanel
    local titleCardCorner = Instance.new("UICorner")
    titleCardCorner.CornerRadius = UDim.new(0, 8)
    titleCardCorner.Parent = titleCard
    
    local titleText = Instance.new("TextLabel")
    titleText.Size = UDim2.new(1, -20, 1, 0)
    titleText.Position = UDim2.new(0, 10, 0, 0)
    titleText.BackgroundTransparency = 1
    titleText.Text = "选择样式"
    titleText.TextColor3 = textColor
    titleText.TextSize = 18
    titleText.Font = Enum.Font.GothamBold
    titleText.TextXAlignment = Enum.TextXAlignment.Left
    titleText.Parent = titleCard
    
    local scrollFrame = Instance.new("ScrollingFrame")
    scrollFrame.Size = UDim2.new(1, -20, 0, 300)
    scrollFrame.Position = UDim2.new(0, 10, 0, 65)
    scrollFrame.BackgroundTransparency = 1
    scrollFrame.BorderSizePixel = 0
    scrollFrame.ScrollBarThickness = 4
    scrollFrame.ScrollBarImageTransparency = 0.5
    scrollFrame.CanvasSize = UDim2.new(0, 0, 0, 0)
    scrollFrame.AutomaticCanvasSize = Enum.AutomaticSize.Y
    scrollFrame.Parent = stylePanel
    
    local btnContainer = Instance.new("Frame")
    btnContainer.Size = UDim2.new(1, 0, 0, 0)
    btnContainer.BackgroundTransparency = 1
    btnContainer.Parent = scrollFrame
    
    local flowLayout = Instance.new("UIListLayout")
    flowLayout.FillDirection = Enum.FillDirection.Vertical
    flowLayout.HorizontalAlignment = Enum.HorizontalAlignment.Center
    flowLayout.Padding = UDim.new(0, 12)
    flowLayout.Parent = btnContainer
    
    styleBtns = {}
    for i, style in ipairs(stylesData) do
        local btn = Instance.new("TextButton")
        btn.Size = UDim2.new(0.9, 0, 0, 80)
        btn.BackgroundColor3 = cardColor
        btn.Text = ""
        btn.BorderSizePixel = 0
        btn.Parent = btnContainer
        local btnCorner = Instance.new("UICorner")
        btnCorner.CornerRadius = UDim.new(0, 10)
        btnCorner.Parent = btn
        
        -- 预览区域（展示样式配色）

        previewContainer.BackgroundTransparency = 1

        previewMain.BackgroundColor3 = style.mainBg
        previewMain.BorderSizePixel = 0
        previewMain.Parent = previewContainer
        local previewMainCorner = Instance.new("UICorner")
        previewMainCorner.CornerRadius = UDim.new(0, 4)
        previewMainCorner.Parent = previewMain
        
        local previewAccent = Instance.new("Frame")
        previewAccent.Size = UDim2.new(1, 0, 0.5, -2)
        previewAccent.Position = UDim2.new(0, 0, 0.5, 2)
        previewAccent.BackgroundColor3 = style.accentColor
        previewAccent.BorderSizePixel = 0
        previewAccent.Parent = previewContainer
        local previewAccentCorner = Instance.new("UICorner")
        previewAccentCorner.CornerRadius = UDim.new(0, 4)
        previewAccentCorner.Parent = previewAccent
        
        local nameLabel = Instance.new("TextLabel")
        nameLabel.Size = UDim2.new(1, -70, 1, 0)
        nameLabel.Position = UDim2.new(0, 70, 0, 0)
        nameLabel.BackgroundTransparency = 1
        nameLabel.Text = style.name
        nameLabel.TextColor3 = textColor
        nameLabel.TextSize = 16
        nameLabel.Font = Enum.Font.GothamSemibold
        nameLabel.TextXAlignment = Enum.TextXAlignment.Left
        nameLabel.Parent = btn
        
        local layoutLabel = Instance.new("TextLabel")
        layoutLabel.Size = UDim2.new(1, -70, 0, 25)
        layoutLabel.Position = UDim2.new(0, 70, 0.5, -10)
        layoutLabel.BackgroundTransparency = 1
        local layoutText = style.layout == "left" and "侧边栏：左侧" or (style.layout == "right" and "侧边栏：右侧" or "侧边栏：顶部")
        layoutLabel.Text = layoutText
        layoutLabel.TextColor3 = Color3.fromRGB(
            textColor.R * 0.7,
            textColor.G * 0.7,
            textColor.B * 0.7
        )
        layoutLabel.TextSize = 12
        layoutLabel.Font = Enum.Font.Gotham
        layoutLabel.TextXAlignment = Enum.TextXAlignment.Left
        layoutLabel.Parent = btn
        
        styleBtns[i] = {btn = btn, data = style}
        
        btn.MouseButton1Click:Connect(function()
            currentStyleIndex = i
            applyFullStyle(i)
        end)
        
        -- 悬停动画效果 [citation:3]
        btn.MouseEnter:Connect(function()
            TweenService:Create(btn, TweenInfo.new(0.2, Enum.EasingStyle.Exponential, Enum.EasingDirection.Out), {
                BackgroundColor3 = style.buttonHoverColor or style.accentColor
            }):Play()
        end)
        btn.MouseLeave:Connect(function()
            TweenService:Create(btn, TweenInfo.new(0.2, Enum.EasingStyle.Exponential, Enum.EasingDirection.Out), {
                BackgroundColor3 = cardColor
            }):Play()
        end)
    end
    
    -- 更新容器高度
    local function updateContainerHeight()
        local totalHeight = 0
        for _, child in ipairs(btnContainer:GetChildren()) do
            if child:IsA("TextButton") then
                totalHeight = totalHeight + child.AbsoluteSize.Y + 12
            end
        end
        btnContainer.Size = UDim2.new(1, 0, 0, totalHeight)
    end
    task.wait(0.1)
    updateContainerHeight()
end

-- ========== 创建侧边栏栏目按钮 ==========
local function rebuildSidebarButtons()
    for _, btn in pairs(tabButtons) do
        if btn then btn:Destroy() end
    end
    tabButtons = {}
    
    local currentStyle = stylesData[currentStyleIndex]
    
    for _, tab in ipairs(tabs) do
        local btn = Instance.new("TextButton")
        btn.Size = UDim2.new(0.85, 0, 0, 42)
        btn.BackgroundColor3 = currentStyle.sidebarBg
        btn.Text = tab.name
        btn.TextColor3 = currentStyle.textColor
        btn.TextSize = 15
        btn.Font = Enum.Font.Gotham
        btn.BorderSizePixel = 0
        btn.Parent = sidebarList
        
        local btnCorner = Instance.new("UICorner")
        btnCorner.CornerRadius = UDim.new(0, 8)
        btnCorner.Parent = btn
        
        tabButtons[tab.id] = btn
        
        btn.MouseButton1Click:Connect(function()
            selectedTab = tab.id
            infoPanel.Visible = (tab.id == "info")
            stylePanel.Visible = (tab.id == "style")
            for id, otherBtn in pairs(tabButtons) do
                if otherBtn then
                    if id == selectedTab then
                        otherBtn.BackgroundColor3 = currentStyle.accentColor
                    else
                        otherBtn.BackgroundColor3 = currentStyle.sidebarBg
                    end
                end
            end
        end)
        
        -- 悬停效果
        btn.MouseEnter:Connect(function()
            if selectedTab ~= tab.id then
                TweenService:Create(btn, TweenInfo.new(0.15), {
                    BackgroundColor3 = currentStyle.buttonHoverColor or currentStyle.accentColor
                }):Play()
            end
        end)
        btn.MouseLeave:Connect(function()
            if selectedTab ~= tab.id then
                TweenService:Create(btn, TweenInfo.new(0.15), {
                    BackgroundColor3 = currentStyle.sidebarBg
                }):Play()
            end
        end)
    end
    
    if tabButtons["info"] then
        tabButtons["info"].BackgroundColor3 = currentStyle.accentColor
    end
end

-- ========== 拖拽系统（基于 Roblox 开发者论坛教程 [citation:1][citation:5][citation:9]） ==========
-- 主窗口拖拽变量
local dragging = false
local dragStart = nil
local startPos = nil

-- 小方块拖拽变量
local miniDragging = false
local miniDragStart = nil
local miniStartPos = nil
local longPressActive = false
local longPressTimer = nil
local LONG_PRESS_TIME = 0.35

-- 获取鼠标位置（兼容鼠标和触摸）
local function getInputPosition(input)
    return input.Position
end

-- 主窗口拖拽更新
local function updateDrag(input)
    if not dragging or not dragStart then return end
    local currentPos = getInputPosition(input)
    local delta = currentPos - dragStart
    local newX = startPos.X.Offset + delta.X
    local newY = startPos.Y.Offset + delta.Y
    
    -- 边界限制
    local maxX = game.CoreGui.AbsoluteSize.X - mainFrame.AbsoluteSize.X
    local maxY = game.CoreGui.AbsoluteSize.Y - mainFrame.AbsoluteSize.Y
    newX = math.clamp(newX, 0, maxX)
    newY = math.clamp(newY, 0, maxY)
    
    mainFrame.Position = UDim2.new(0, newX, 0, newY)
end

-- 小方块拖拽更新
local function updateMiniDrag(input)
    if not miniDragging or not miniDragStart then return end
    local currentPos = getInputPosition(input)
    local delta = currentPos - miniDragStart
    local newX = miniStartPos.X.Offset + delta.X
    local newY = miniStartPos.Y.Offset + delta.Y
    
    local maxX = game.CoreGui.AbsoluteSize.X - miniFrame.AbsoluteSize.X
    local maxY = game.CoreGui.AbsoluteSize.Y - miniFrame.AbsoluteSize.Y
    newX = math.clamp(newX, 0, maxX)
    newY = math.clamp(newY, 0, maxY)
    
    miniFrame.Position = UDim2.new(0, newX, 0, newY)
end

-- 结束所有拖拽
local function endAllDrag()
    dragging = false
    dragStart = nil
    startPos = nil
    miniDragging = false
    miniDragStart = nil
    miniStartPos = nil
    if longPressTimer then
        longPressTimer:Disconnect()
        longPressTimer = nil
    end
    longPressActive = false
end

-- 标题栏拖拽（直接拖拽，无需长按）
titleBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = getInputPosition(input)
        startPos = mainFrame.Position
    end
end)

-- 小方块：长按拖拽 + 单击恢复
local miniFrameInputBegan
miniFrameInputBegan = miniFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        longPressActive = true
        longPressTimer = task.delay(LONG_PRESS_TIME, function()
            if longPressActive and miniFrame.Visible then
                miniDragging = true
                miniDragStart = getInputPosition(input)
                miniStartPos = miniFrame.Position
                longPressActive = false
            end
        end)
    end
end)

-- 小方块输入结束
miniFrame.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        if longPressTimer then
            longPressTimer:Cancel()
            longPressTimer = nil
        end
        
        if miniDragging then
            -- 长按拖拽结束，不恢复窗口
            miniDragging = false
            miniDragStart = nil
            miniStartPos = nil
        elseif longPressActive and miniFrame.Visible then
            -- 单击：恢复主窗口
            mainFrame.Visible = true
            miniFrame.Visible = false
            mainFrame.Position = UDim2.new(0, miniFrame.AbsolutePosition.X, 0, miniFrame.AbsolutePosition.Y)
        end
        longPressActive = false
    end
end)

-- 全局输入更新
UserInputService.InputChanged:Connect(function(input)
    if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        updateDrag(input)
    elseif miniDragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        updateMiniDrag(input)
    end
end)

-- 确保松开按钮时结束拖拽
UserInputService.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        endAllDrag()
    end
end)
