-- ==================== 辅助函数 ====================
local function getCenterPosition(guiObject)
    local absPos = guiObject.AbsolutePosition
    local absSize = guiObject.AbsoluteSize
    return Vector2.new(absPos.X + absSize.X / 2, absPos.Y + absSize.Y / 2)
end

local function setPositionFromCenter(guiObject, centerPos)
    local size = guiObject.AbsoluteSize
    local newPos = Vector2.new(centerPos.X - size.X / 2, centerPos.Y - size.Y / 2)
    guiObject.Position = UDim2.new(0, newPos.X, 0, newPos.Y)
end

-- 圆点脉冲动画
local pulseTween
local function startPulse()
    if pulseTween then pulseTween:Cancel() end
    pulseTween = TweenService:Create(dot, TweenInfo.new(0.8, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut, -1, true), 
        {BackgroundTransparency = 0.3})
    pulseTween:Play()
end
local function stopPulse()
    if pulseTween then pulseTween:Cancel() end
    dot.BackgroundTransparency = 0.1
end

-- ==================== 拖拽逻辑（悬浮窗） ====================
local windowDrag = { dragging = false, startPos = nil, startMouse = nil }

titleBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        windowDrag.dragging = true
        windowDrag.startPos = floatingWindow.Position
        windowDrag.startMouse = input.Position
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                windowDrag.dragging = false
            end
        end)
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if windowDrag.dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        local delta = input.Position - windowDrag.startMouse
        floatingWindow.Position = UDim2.new(0, windowDrag.startPos.X.Offset + delta.X, 0, windowDrag.startPos.Y.Offset + delta.Y)
    end
end)

UserInputService.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        windowDrag.dragging = false
    end
end)

-- ==================== 圆点拖动与单击恢复（区分长按） ====================
local dotDrag = {
    active = false,
    isDragging = false,
    pressTime = 0,
    startPos = nil,
    startMouse = nil,
    connectionChanged = nil,
    connectionEnded = nil,
    longPressTimer = nil,
}

local function cleanupDotDrag()
    if dotDrag.connectionChanged then dotDrag.connectionChanged:Disconnect() end
    if dotDrag.connectionEnded then dotDrag.connectionEnded:Disconnect() end
    if dotDrag.longPressTimer then task.cancel(dotDrag.longPressTimer) end
    dotDrag.active = false
    dotDrag.isDragging = false
    dotDrag.pressTime = 0
end

local function startDotDragProcess(startInput)
    if dotDrag.active then return end
    dotDrag.active = true
    dotDrag.isDragging = false
    dotDrag.startPos = dot.Position
    dotDrag.startMouse = startInput.Position
    dotDrag.pressTime = tick()

    -- 长按定时器：长按后进入拖动模式（允许移动）
    dotDrag.longPressTimer = task.delay(longPressDuration, function()
        if dotDrag.active and not dotDrag.isDragging then
            dotDrag.isDragging = true
            -- 可选：视觉反馈，改变圆点透明度
            TweenService:Create(dot, TweenInfo.new(0.2), {BackgroundTransparency = 0.3}):Play()
        end
    end)

    -- 监听移动
    dotDrag.connectionChanged = UserInputService.InputChanged:Connect(function(input)
        if not dotDrag.active then return end
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            local delta = input.Position - dotDrag.startMouse
            local distance = math.sqrt(delta.X^2 + delta.Y^2)
            -- 如果移动超过阈值且尚未进入拖动模式，立即进入拖动模式（无需等待长按）
            if not dotDrag.isDragging and distance >= dragThreshold then
                dotDrag.isDragging = true
                if dotDrag.longPressTimer then task.cancel(dotDrag.longPressTimer) end
                TweenService:Create(dot, TweenInfo.new(0.2), {BackgroundTransparency = 0.3}):Play()
            end
            if dotDrag.isDragging then
                local newX = dotDrag.startPos.X.Offset + delta.X
                local newY = dotDrag.startPos.Y.Offset + delta.Y
                dot.Position = UDim2.new(0, newX, 0, newY)
            end
        end
    end)

    -- 监听结束
    dotDrag.connectionEnded = UserInputService.InputEnded:Connect(function(input)
        if not dotDrag.active then return end
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            local pressDuration = tick() - dotDrag.pressTime
            -- 如果不是拖动模式，且按下时间短于长按阈值，视为单击 → 恢复窗口
            if not dotDrag.isDragging and pressDuration < longPressDuration then
                restoreWindowFromDot()
            end
            -- 恢复圆点透明度
            TweenService:Create(dot, TweenInfo.new(0.2), {BackgroundTransparency = 0.1}):Play()
            cleanupDotDrag()
        end
    end)
end

dot.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        startDotDragProcess(input)
    end
end)

-- ==================== 缩小与恢复函数 ====================
local function minimizeToDot()
    local windowCenter = getCenterPosition(floatingWindow)
    floatingWindow.Visible = false
    dot.Visible = true
    setPositionFromCenter(dot, windowCenter)
    startPulse()
end

local function restoreWindowFromDot()
    local dotCenter = getCenterPosition(dot)
    floatingWindow.Visible = true
    dot.Visible = false
    setPositionFromCenter(floatingWindow, dotCenter)
    stopPulse()
    -- 窗口淡入效果
    floatingWindow.BackgroundTransparency = 0.08
end
