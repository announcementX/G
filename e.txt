-- ========== 应用完整样式 ==========
local function applyFullStyle()
    local style = stylesData[currentStyleIndex]
    if not style then return end
    
    mainFrame.BackgroundColor3 = style.mainBg
    sidebar.BackgroundColor3 = style.sidebarBg
    titleBar.BackgroundColor3 = style.titleBg
    contentArea.BackgroundColor3 = style.contentBg
    minBtn.BackgroundColor3 = style.sidebarBg
    titleLabel.TextColor3 = style.textColor
    clockLabel.TextColor3 = style.textColor
    
    buildInfoPanel(style)
    buildStylePanel()
    
    -- 布局
    local sidebarWidth = 140
    if style.layout == "left" then
        sidebar.Size = UDim2.new(0, sidebarWidth, 1, -40)
        sidebar.Position = UDim2.new(0, 0, 0, 40)
        sidebarList.Size = UDim2.new(1, 0, 1, 0)
        listLayout.FillDirection = Enum.FillDirection.Vertical
        contentArea.Size = UDim2.new(1, -sidebarWidth, 1, -40)
        contentArea.Position = UDim2.new(0, sidebarWidth, 0, 40)
    elseif style.layout == "right" then
        sidebar.Size = UDim2.new(0, sidebarWidth, 1, -40)
        sidebar.Position = UDim2.new(1, -sidebarWidth, 0, 40)
        sidebarList.Size = UDim2.new(1, 0, 1, 0)
        listLayout.FillDirection = Enum.FillDirection.Vertical
        contentArea.Size = UDim2.new(1, -sidebarWidth, 1, -40)
        contentArea.Position = UDim2.new(0, 0, 0, 40)
    elseif style.layout == "top" then
        sidebar.Size = UDim2.new(1, 0, 0, 60)
        sidebar.Position = UDim2.new(0, 0, 0, 40)
        sidebarList.Size = UDim2.new(1, 0, 1, 0)
        listLayout.FillDirection = Enum.FillDirection.Horizontal
        listLayout.HorizontalAlignment = Enum.HorizontalAlignment.Center
        listLayout.Padding = UDim.new(0, 20)
        contentArea.Size = UDim2.new(1, 0, 1, -100)
        contentArea.Position = UDim2.new(0, 0, 0, 100)
    end
    
    for id, btn in pairs(tabButtons) do
        if btn then
            btn.BackgroundColor3 = style.sidebarBg
            btn.TextColor3 = style.textColor
            if selectedTab == id then
                btn.BackgroundColor3 = style.accentColor
            end
        end
    end
    
    for i, btn in ipairs(styleBtns) do
        if btn then
            if i == currentStyleIndex then
                TweenService:Create(btn, TweenInfo.new(0.2), {BackgroundColor3 = style.accentColor}):Play()
            else
                TweenService:Create(btn, TweenInfo.new(0.2), {BackgroundColor3 = style.cardBg}):Play()
            end
        end
    end
end

-- ========== 构建侧边栏按钮 ==========
local function buildSidebarButtons()
    for _, v in pairs(tabButtons) do if v then v:Destroy() end end
    tabButtons = {}
    local style = stylesData[currentStyleIndex]
    for _, tab in ipairs(tabs) do
        local btn = Instance.new("TextButton")
        btn.Size = UDim2.new(0.85, 0, 0, 42)
        btn.BackgroundColor3 = style.sidebarBg
        btn.Text = tab.name
        btn.TextColor3 = style.textColor
        btn.TextSize = 15
        btn.Font = Enum.Font.Gotham
        btn.BorderSizePixel = 0
        btn.Parent = sidebarList
        local corner = Instance.new("UICorner")
        corner.CornerRadius = UDim.new(0, 8)
        corner.Parent = btn
        tabButtons[tab.id] = btn
        btn.MouseButton1Click:Connect(function()
            selectedTab = tab.id
            infoPanel.Visible = (tab.id == "info")
            stylePanel.Visible = (tab.id == "style")
            for id, b in pairs(tabButtons) do
                if b then
                    b.BackgroundColor3 = (id == selectedTab) and style.accentColor or style.sidebarBg
                end
            end
        end)
        btn.MouseEnter:Connect(function()
            if selectedTab ~= tab.id then
                TweenService:Create(btn, TweenInfo.new(0.15), {BackgroundColor3 = style.hoverColor}):Play()
            end
        end)
        btn.MouseLeave:Connect(function()
            if selectedTab ~= tab.id then
                TweenService:Create(btn, TweenInfo.new(0.15), {BackgroundColor3 = style.sidebarBg}):Play()
            end
        end)
    end
    if tabButtons["info"] then
        tabButtons["info"].BackgroundColor3 = style.accentColor
    end
end

-- ========== 拖拽系统 ==========
local dragData = { active = false, frame = nil, startPos = nil, startMouse = nil }
local miniDragData = { active = false, startPos = nil, startMouse = nil, longPressActive = false, timer = nil }

local function startDrag(frame, input)
    dragData.active = true
    dragData.frame = frame
    dragData.startPos = frame.Position
    dragData.startMouse = input.Position
end

local function updateDrag(input)
    if not dragData.active then return end
    local delta = input.Position - dragData.startMouse
    local newX = dragData.startPos.X.Offset + delta.X
    local newY = dragData.startPos.Y.Offset + delta.Y
    local maxX = game.CoreGui.AbsoluteSize.X - dragData.frame.AbsoluteSize.X
    local maxY = game.CoreGui.AbsoluteSize.Y - dragData.frame.AbsoluteSize.Y
    dragData.frame.Position = UDim2.new(0, math.clamp(newX, 0, maxX), 0, math.clamp(newY, 0, maxY))
end

titleBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        startDrag(mainFrame, input)
    end
end)

miniFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        miniDragData.longPressActive = true
        miniDragData.timer = task.delay(0.35, function()
            if miniDragData.longPressActive and miniFrame.Visible then
                miniDragData.active = true
                miniDragData.startPos = miniFrame.Position
                miniDragData.startMouse = input.Position
                miniDragData.longPressActive = false
            end
        end)
    end
end)

miniFrame.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        if miniDragData.timer then miniDragData.timer:Cancel() end
        if miniDragData.active then
            miniDragData.active = false
        elseif miniDragData.longPressActive and miniFrame.Visible then
            mainFrame.Visible = true
            miniFrame.Visible = false
            mainFrame.Position = UDim2.new(0, miniFrame.AbsolutePosition.X, 0, miniFrame.AbsolutePosition.Y)
        end
        miniDragData.longPressActive = false
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if dragData.active and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        updateDrag(input)
    elseif miniDragData.active and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        local delta = input.Position - miniDragData.startMouse
        local newX = miniDragData.startPos.X.Offset + delta.X
        local newY = miniDragData.startPos.Y.Offset + delta.Y
        local maxX = game.CoreGui.AbsoluteSize.X - miniFrame.AbsoluteSize.X
        local maxY = game.CoreGui.AbsoluteSize.Y - miniFrame.AbsoluteSize.Y
        miniFrame.Position = UDim2.new(0, math.clamp(newX, 0, maxX), 0, math.clamp(newY, 0, maxY))
    end
end)

UserInputService.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragData.active = false
        miniDragData.active = false
    end
end)
