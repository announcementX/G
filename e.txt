local CoreGui = game:GetService("CoreGui")
local TweenService = game:GetService("TweenService")
local RunService = game:GetService("RunService")
local UserInputService = game:GetService("UserInputService")

-- 1. 炫酷加载动画 (XU Logo)
local function StartupAnimation()
    local ScreenGui = Instance.new("ScreenGui", CoreGui)
    local Bg = Instance.new("Frame", ScreenGui)
    Bg.Size = UDim2.new(1, 0, 1, 0)
    Bg.BackgroundColor3 = Color3.fromRGB(10, 10, 10)
    
    local Logo = Instance.new("TextLabel", Bg)
    Logo.Text = "XU"
    Logo.Size = UDim2.new(0, 200, 0, 100)
    Logo.Position = UDim2.new(0.5, -100, 0.4, -50)
    Logo.TextColor3 = Color3.fromRGB(30, 215, 96) -- 绿色
    Logo.Font = Enum.Font.GothamBold
    Logo.TextSize = 100
    Logo.BackgroundTransparency = 1
    
    local Hint = Instance.new("TextLabel", Bg)
    Hint.Text = "正在加载XU脚本..."
    Hint.Position = UDim2.new(0.5, -100, 0.6, 0)
    Hint.Size = UDim2.new(0, 200, 30)
    Hint.TextColor3 = Color3.fromRGB(200, 200, 200)
    Hint.BackgroundTransparency = 1

    -- 呼吸效果
    local ti = TweenInfo.new(1, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut, -1, true)
    TweenService:Create(Logo, ti, {TextTransparency = 0.5}):Play()
    
    task.wait(3)
    ScreenGui:Destroy()
end

-- 2. 主悬浮窗逻辑
local MainGui = Instance.new("ScreenGui", CoreGui)
local MainFrame = Instance.new("Frame", MainGui)
local UICorner = Instance.new("UICorner", MainFrame)
local Sidebar = Instance.new("Frame", MainFrame)
local Content = Instance.new("Frame", MainFrame)
local TitleBar = Instance.new("Frame", MainFrame)
local TimeLabel = Instance.new("TextLabel", TitleBar)

-- 基础设置
MainFrame.Size = UDim2.new(0, 450, 0, 280)
MainFrame.Position = UDim2.new(0.5, -225, 0.5, -140)
MainFrame.BackgroundColor3 = Color3.fromRGB(18, 18, 18)
MainFrame.ClipsDescendants = true
UICorner.CornerRadius = UDim.new(0, 12)

-[span_1](start_span)- 北京时间逻辑[span_1](end_span)
RunService.Heartbeat:Connect(function()
    TimeLabel.Text = "北京时间: " .. os.date("%H:%M:%S")
end)
TimeLabel.Size = UDim2.new(0, 200, 1, 0)
TimeLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
TimeLabel.BackgroundTransparency = 1
TimeLabel.Font = Enum.Font.Gotham

-- 3. 缩小与长按移动逻辑
local Minimized = false
local MinimizeBtn = Instance.new("TextButton", TitleBar)
MinimizeBtn.Text = "X"
MinimizeBtn.Size = UDim2.new(0, 30, 0, 30)
MinimizeBtn.Position = UDim2.new(1, -35, 0, 5)
MinimizeBtn.BackgroundColor3 = Color3.fromRGB(255, 50, 50)

local function ToggleMinimize()
    Minimized = not Minimized
    if Minimized then
        TweenService:Create(MainFrame, TweenInfo.new(0.4), {Size = UDim2.new(0, 60, 0, 60)}):Play()
        Sidebar.Visible = false
        Content.Visible = false
        MinimizeBtn.Text = "XU"
    else
        TweenService:Create(MainFrame, TweenInfo.new(0.4), {Size = UDim2.new(0, 450, 0, 280)}):Play()
        Sidebar.Visible = true
        Content.Visible = true
        MinimizeBtn.Text = "X"
    end
end
MinimizeBtn.MouseButton1Click:Connect(ToggleMinimize)

-- 防误触长按移动
local Holding = false
local StartTime = 0
MainFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        Holding = true
        StartTime = tick()
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if Holding and (tick() - StartTime > 0.5) then -- 长按超过0.5秒开启移动
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            MainFrame.Position = UDim2.new(0, input.Position.X - 30, 0, input.Position.Y - 15)
        end
    end
end)

MainFrame.InputEnded:Connect(function() Holding = false end)

-- 4. 栏目与样式系统
local function CreateTab(name, parent)
    local btn = Instance.new("TextButton", parent)
    btn.Text = name
    btn.Size = UDim2.new(1, -10, 0, 40)
    btn.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
    btn.TextColor3 = Color3.new(1,1,1)
    Instance.new("UICorner", btn)
    return btn
end

-- 信息栏目
local InfoBtn = CreateTab("信息栏目", Sidebar)
InfoBtn.MouseButton1Click:Connect(function()
    -- 清空并显示信息
    Content:ClearAllChildren()
    local txt = Instance.new("TextLabel", Content)
    txt.Size = UDim2.new(1, 0, 1, 0)
    txt.Text = "作者: HaoChen\nQQ: 1626844714"
    txt.TextColor3 = Color3.new(1,1,1)
    txt.BackgroundTransparency = 1
end)

-- 样式切换系统
local StyleBtn = CreateTab("样式栏目", Sidebar)
StyleBtn.MouseButton1Click:Connect(function()
    Content:ClearAllChildren()
    local s1 = CreateTab("Spotify绿", Content)
    s1.MouseButton1Click:Connect(function()
        MainFrame.BackgroundColor3 = Color3.fromRGB(18, 18, 18)
        Sidebar.BackgroundColor3 = Color3.fromRGB(25, 25, 25)
    end)
    
    local s2 = CreateTab("深海蓝(换布局)", Content)
    s2.MouseButton1Click:Connect(function()
        MainFrame.BackgroundColor3 = Color3.fromRGB(10, 20, 35)
        Sidebar.Position = UDim2.new(1, -130, 0, 40) -- 侧边栏移到右侧
    end)
end)

-- 初始化启动
coroutine.wrap(StartupAnimation)()
