-- 完整 XU飞行 GUI

local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local StarterGui = game:GetService("StarterGui")

local player = Players.LocalPlayer
local char = player.Character or player.CharacterAdded:Wait()
local hum = char:FindFirstChildWhichIsA("Humanoid")

-- UI 元素
local main = Instance.new("ScreenGui")
main.Name = "XUFlightGUI"
main.Parent = player:WaitForChild("PlayerGui")
main.ZIndexBehavior = Enum.ZIndexBehavior.Sibling
main.ResetOnSpawn = false

local Frame = Instance.new("Frame")
Frame.Name = "Frame"
Frame.Parent = main
Frame.BackgroundColor3 = Color3.fromRGB(10,10,30)
Frame.Position = UDim2.new(0.1,0,0.35,0)
Frame.Size = UDim2.new(0,260,0,150)
Frame.Active = true
Frame.Draggable = true

-- 按钮和标签
local up = Instance.new("TextButton")
up.Name = "up"; up.Parent = Frame
up.BackgroundColor3 = Color3.fromRGB(20,20,50); up.Size = UDim2.new(0,44,0,28)
up.Font = Enum.Font.SourceSans; up.Text = "上"; up.TextColor3 = Color3.fromRGB(200,200,255); up.TextSize = 14

local down = Instance.new("TextButton")
down.Name = "down"; down.Parent = Frame
down.BackgroundColor3 = Color3.fromRGB(20,20,50); down.Size = UDim2.new(0,44,0,28); down.Position = UDim2.new(0,0,0.33,0)
down.Font = Enum.Font.SourceSans; down.Text = "下"; down.TextColor3 = Color3.fromRGB(200,200,255); down.TextSize = 14

local onof = Instance.new("TextButton")
onof.Name = "onof"; onof.Parent = Frame
onof.BackgroundColor3 = Color3.fromRGB(20,20,50); onof.Size = UDim2.new(0,56,0,28); onof.Position = UDim2.new(0.7,0,0.33,0)
onof.Font = Enum.Font.SourceSans; onof.Text = "飞行"; onof.TextColor3 = Color3.fromRGB(200,200,255); onof.TextSize = 14

local plus = Instance.new("TextButton")
plus.Name = "plus"; plus.Parent = Frame
plus.BackgroundColor3 = Color3.fromRGB(20,20,50); plus.Size = UDim2.new(0,45,0,28); plus.Position = UDim2.new(0.25,0,0,0)
plus.Font = Enum.Font.SourceSans; plus.Text = "+"; plus.TextColor3 = Color3.fromRGB(200,200,255); plus.TextScaled = true; plus.TextWrapped = true

local mine = Instance.new("TextButton")
mine.Name = "mine"; mine.Parent = Frame
mine.BackgroundColor3 = Color3.fromRGB(20,20,50); mine.Size = UDim2.new(0,45,0,28); mine.Position = UDim2.new(0.25,0,0.33,0)
mine.Font = Enum.Font.SourceSans; mine.Text = "-"; mine.TextColor3 = Color3.fromRGB(200,200,255); mine.TextScaled = true; mine.TextWrapped = true

local speed = Instance.new("TextLabel")
speed.Name = "speed"; speed.Parent = Frame
speed.BackgroundColor3 = Color3.fromRGB(20,20,50); speed.Size = UDim2.new(0,44,0,28); speed.Position = UDim2.new(0.45,0,0.33,0)
speed.Font = Enum.Font.SourceSans; speed.Text = "1"; speed.TextColor3 = Color3.fromRGB(200,200,255); speed.TextScaled = true; speed.TextWrapped = true

-- 关闭 & 缩小按钮
local closebutton = Instance.new("TextButton")
closebutton.Name = "Close"; closebutton.Parent = Frame
closebutton.BackgroundColor3 = Color3.fromRGB(20,20,50); closebutton.Size = UDim2.new(0,45,0,28); closebutton.Position = UDim2.new(0,0,-0.2,27)
closebutton.Font = Enum.Font.SourceSans; closebutton.Text = "关闭"; closebutton.TextSize = 20; closebutton.TextColor3 = Color3.fromRGB(255,180,180)

local mini = Instance.new("TextButton")
mini.Name = "minimize"; mini.Parent = Frame
mini.BackgroundColor3 = Color3.fromRGB(20,20,50); mini.Size = UDim2.new(0,45,0,28); mini.Position = UDim2.new(0,50,-0.2,27)
mini.Font = Enum.Font.SourceSans; mini.Text = "缩小"; mini.TextSize = 20; mini.TextColor3 = Color3.fromRGB(180,255,255)

local mini2 = Instance.new("TextButton")
mini2.Name = "minimize2"; mini2.Parent = Frame
mini2.BackgroundColor3 = Color3.fromRGB(10,10,30); mini2.Size = UDim2.new(0,180,0,50); mini2.Position = UDim2.new(0,0,0,0)
mini2.Font = Enum.Font.GothamBold; mini2.Text = "🌌 XU飞行"; mini2.TextScaled = true
mini2.TextColor3 = Color3.fromRGB(180,220,255); mini2.Visible = false

local speeds = 1
local nowe = false

-- 飞行逻辑
onof.MouseButton1Down:Connect(function()
    nowe = not nowe
    local hum = player.Character:FindFirstChildWhichIsA("Humanoid")
    if nowe then
        hum.PlatformStand = true
    else
        hum.PlatformStand = false
    end
end)

-- 缩小逻辑
mini.MouseButton1Click:Connect(function()
    up.Visible, down.Visible, onof.Visible, plus.Visible, speed.Visible, mine.Visible, mini.Visible = false,false,false,false,false,false,false
    mini2.Visible = true
end)

mini2.MouseButton1Click:Connect(function()
    up.Visible, down.Visible, onof.Visible, plus.Visible, speed.Visible, mine.Visible, mini.Visible = true
    mini2.Visible = false
end)

-- 关闭逻辑
closebutton.MouseButton1Click:Connect(function()
    main:Destroy()
end)

-- 上下移动
local upLoop, downLoop
up.MouseButton1Down:Connect(function()
    upLoop = true
    spawn(function()
        while upLoop do
            RunService.RenderStepped:Wait()
            char.HumanoidRootPart.CFrame = char.HumanoidRootPart.CFrame * CFrame.new(0,1,0)
        end
    end)
end)
up.MouseButton1Up:Connect(function() upLoop = false end)

down.MouseButton1Down:Connect(function()
    downLoop = true
    spawn(function()
        while downLoop do
            RunService.RenderStepped:Wait()
            char.HumanoidRootPart.CFrame = char.HumanoidRootPart.CFrame * CFrame.new(0,-1,0)
        end
    end)
end)
down.MouseButton1Up:Connect(function() downLoop = false end)

-- 速度加减
plus.MouseButton1Down:Connect(function()
    speeds = speeds + 1
    speed.Text = tostring(speeds)
end)
mine.MouseButton1Down:Connect(function()
    if speeds > 1 then
        speeds = speeds - 1
        speed.Text = tostring(speeds)
    end
end)

-- 飞行移动核心（简化 R6/R15 支持）
spawn(function()
    local ctrl = {f=0,b=0,l=0,r=0}
    local lastctrl = {f=0,b=0,l=0,r=0}
    local maxspeed = 50
    local speed = 0
    while true do
        RunService.RenderStepped:Wait()
        if nowe then
            local torso = char:FindFirstChild("Torso") or char:FindFirstChild("UpperTorso")
            if torso then
                local bg = torso:FindFirstChild("BodyGyro") or Instance.new("BodyGyro",torso)
                bg.P=9e4; bg.maxTorque=Vector3.new(9e9,9e9,9e9); bg.CFrame=torso.CFrame
                local bv = torso:FindFirstChild("BodyVelocity") or Instance.new("BodyVelocity",torso)
                bv.velocity=Vector3.new(0,0.1,0); bv.maxForce=Vector3.new(9e9,9e9,9e9)
            end
        end
    end
end)

-- 通知
StarterGui:SetCore("SendNotification",{
    Title="XU飞行";
    Text="HaoChen";
    Duration=5;
})
