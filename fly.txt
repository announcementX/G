-- 飞行
local flying = false
local speed = 1
local vertical = 0
local modeType = 1
local bv,bg

local function startFly()
	flying = true
	flyBtn.Text = "停止"

	local root = player.Character:WaitForChild("HumanoidRootPart")

	bv = Instance.new("BodyVelocity", root)
	bv.MaxForce = Vector3.new(9e9,9e9,9e9)

	bg = Instance.new("BodyGyro", root)
	bg.MaxTorque = Vector3.new(9e9,9e9,9e9)

	RunService:BindToRenderStep("fly",0,function()
		if not flying then return end

		local cam = workspace.CurrentCamera

		if modeType == 1 then
			-- 视角模式（你现在要的）
			bv.Velocity = cam.CFrame.LookVector * speed * 5
			bg.CFrame = cam.CFrame

		else
			-- 原始模式（传统飞行）
			local dir = Vector3.new(0,vertical,0)
			bv.Velocity = dir * speed * 10
		end
	end)
end

local function stopFly()
	flying = false
	flyBtn.Text = "开始"
	RunService:UnbindFromRenderStep("fly")
	if bv then bv:Destroy() end
	if bg then bg:Destroy() end
end

-- 控制
flyBtn.MouseButton1Click:Connect(function()
	if flying then stopFly() else startFly() end
end)

plus.MouseButton1Click:Connect(function()
	speed += 1
	speedText.Text = "SPEED "..speed
end)

minus.MouseButton1Click:Connect(function()
	if speed > 1 then speed -= 1 end
	speedText.Text = "SPEED "..speed
end)

up.MouseButton1Down:Connect(function() vertical = 1 end)
up.MouseButton1Up:Connect(function() vertical = 0 end)

down.MouseButton1Down:Connect(function() vertical = -1 end)
down.MouseButton1Up:Connect(function() vertical = 0 end)

-- 模式切换
mode.MouseButton1Click:Connect(function()
	modeType = modeType == 1 and 2 or 1
	mode.Text = modeType == 1 and "模式1" or "模式2"
end)

-- 缩小逻辑
minimize.MouseButton1Click:Connect(function()
	frame.Visible = false
	mini.Visible = true
end)

mini.MouseButton1Click:Connect(function()
	frame.Visible = true
	mini.Visible = false
end)
